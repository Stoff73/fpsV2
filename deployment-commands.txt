# TenGo v0.2.13 Deployment Commands
# ===================================

# 1. Upload the tar file to the server
scp v0.2.13-deployment.tar.gz YOUR_USERNAME@YOUR_SERVER:/home/YOUR_USERNAME/

# 2. SSH into the server
ssh YOUR_USERNAME@YOUR_SERVER

# 3. Extract the tar file in the tengo-app directory
cd ~/tengo-app
tar -xzf ~/v0.2.13-deployment.tar.gz

# 4. Verify extraction
ls -la database/migrations/2025_11_24_*.php
ls -la app/Services/UserProfile/UserProfileService.php

# 5. Run migrations
php artisan migrate --step

# 6. Rebuild frontend
export NODE_ENV=production
npm run build

# 7. Clear all caches
php artisan cache:clear
php artisan config:clear
php artisan view:clear
php artisan route:clear

# 8. Verify deployment
php artisan --version
php artisan migrate:status

# 9. Clean up
rm ~/v0.2.13-deployment.tar.gz
